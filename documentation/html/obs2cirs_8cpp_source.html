51 std::map<std::string, double> defaultoptions()
52 {
53  // Define the default values of some of the unnecessary parameters
54  std::map<std::string, double> options ;
55  options["dut1"] = 0.0 ;
56  options["xpolar"] = 0.0 ;
57  options["ypolar"] = 0.0 ;
58 
59  // Set the current time as the default date for query
60  time_t current_time = time(NULL) ;
61  struct tm * timeinfo = gmtime(&current_time) ;
62  double dayfrac = (timeinfo->tm_hour + (timeinfo->tm_min/60.0) + ((timeinfo->tm_sec)/3600.0))/24.0 ;
63 // std::vector<double> date_vect(4,0.0);
64 // date_vect[0] = {double(timeinfo->tm_year+1900.0), double(timeinfo->tm_mon+1.0), double(1.0*timeinfo->tm_mday), dayfrac} ;
65  CEDate date({timeinfo->tm_year+1900.0, timeinfo->tm_mon+1.0, 1.0*timeinfo->tm_mday, dayfrac}) ;
66 
67  options["juliandate"] = date.JD() ;
68 
69  // Set the default observer weather conditions
70  CEObserver observer ;
71  options["elevation"] = observer.Elevation() ;
72  options["humidity"] = observer.RelativeHumidity() ;

100 std::map<std::string, double> parseoptions(int argc, char** argv, const struct option* longopts)
101 {
102  // Setup the default parameters
103  std::map<std::string, double> options = defaultoptions() ;
104 
105  int c;
106 
107  while (1)
108  {
109  /* getopt_long stores the option index here. */
110  int option_index = 0;
111 
112  c = getopt_long (argc, argv, "hX:Y:e:R:D:j:p:t:r:w:x:y:",
113  longopts, &option_index);
114 
115  /* Detect the end of the options. */
116  if (c == -1) break;
117 
118  switch (c)
119  {
120  case 0:
121  /* If this option set a flag, do nothing else now. */
122  if (longopts[option_index].flag != 0)
123  break;
124 
125  // Set the option
126  options[longopts[option_index].name] = std::stod(optarg) ;
127  break;
128 
129  case 'h':
130  // Print the help message and quit
131  Print_Help() ;
132  return std::map<std::string, double>() ;
133  case 'X': // Assign observer's Longitude
134  options["longitude"] = std::stod(optarg) ;
135  break;
136  case 'Y': // Assign observer's Latitude
137  options["latitude"] = std::stod(optarg) ;
138  break;
139  case 'e': // Assign observer's elevation
140  options["elevation"] = std::stod(optarg) ;
141  break;
142  case 'R': // Assign Right Ascension coordinate
143  options["ra"] = std::stod(optarg) ;
144  break;
145  case 'D': // Assign Declination coordinate
146  options["dec"] = std::stod(optarg) ;
147  break;
148  case 'j': // Assign julian date
149  options["juliandate"] = std::stod(optarg) ;
150  break;
151  case 'p':
152  options["pressure"] = std::stod(optarg) ;
153  break;
154  case 't':
155  options["temperature"] = std::stod(optarg) ;
156  break;
157  case 'r':
158  options["humidity"] = std::stod(optarg) ;
159  break;
160  case 'w':
161  options["wavelength"] = std::stod(optarg) ;
162  break;
163  case 'd':
164  options["dut1"] = std::stod(optarg) ;
165  break;
166  case 'x':
167  options["xpolar"] = std::stod(optarg) ;
168  break;
169  case 'y':
170  options["ypolar"] = std::stod(optarg) ;
171  break;
172  case '?':
173  /* getopt_long already printed an error message. */
174  break;
175 
176  default:
177  break ;
178  }
179  }
180 
181 
182  return options ;
183 }
