84 int main(int argc, char** argv) {
85 
86  // Parse the command line options
87  CEExecOptions opts = DefineOpts() ;
88  if (opts.ParseCommandLine(argc, argv)) return 0 ;
89 
90  // Create a map to store the results
91  std::map<std::string, double> results ;
92  results["glon"] = 0.0 ;
93  results["glat"] = 0.0 ;
94 
95  // Define the observer
96  CEObserver observer(opts.AsDouble("longitude"),
97  opts.AsDouble("latitude"),
98  opts.AsDouble("elevation"),
99  CEAngleType::DEGREES);
100  observer.SetPressure_hPa(opts.AsDouble("pressure"));
101  observer.SetTemperature_C(opts.AsDouble("temperature"));
102  observer.SetRelativeHumidity(opts.AsDouble("humidity"));
103  observer.SetWavelength_um(opts.AsDouble("wavelength"));
104 
105  // Define the date
106  CEDate date(opts.AsDouble("juliandate"), CEDateType::JD);
107 
108  // Convert the coordinates
109  int errcode = CECoordinates::Observed2Galactic(opts.AsDouble("azimuth")*DD2R,
110  opts.AsDouble("zenith")*DD2R,
111  &results["glon"], &results["glat"],
112  date, observer, CEAngleType::RADIANS);
113  results["glon"] *= DR2D;
114  results["glat"] *= DR2D;
115 
116  // Print the results
117  PrintResults(opts, results) ;
118 
119  return errcode ;
120 }
