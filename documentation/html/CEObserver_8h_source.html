269 std::vector<double> CEObserver::Time(const CEDate& date)
270 {
271  double utc = date.MJD();
272  utc -= std::floor(utc);
273  utc += utc_offset_/24.0; // Do this here for precision reasons
274  utc *= CppEphem::sec_per_day();
275  return CETime::TimeDbl2Vect( CETime::TimeSec2Time(utc) ) ;
276 }
