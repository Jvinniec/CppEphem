161 bool test_CEObserver::test_set_atmoPars()
162 {

164  double old_tol(DblTol());
165  SetDblTol(1.0e-11);
166

174  obs.SetTemperature_K(temp_K);

176  test_double(obs.Temperature_C(), temp_C, __func__, __LINE__);
177  test_double(obs.Temperature_F(), temp_F, __func__, __LINE__);
178 
179  // Test Temperature setting
180  temp_F += 10.0;
181  obs.SetTemperature_F(temp_F);
182  test_double(obs.Temperature_F(), temp_F, __func__, __LINE__);
183  temp_C -= 10.0;
184  obs.SetTemperature_C(temp_C);
185  test_double(obs.Temperature_C(), temp_C, __func__, __LINE__);
186 
187  // Test pressure
188  double pres = obs.Pressure_hPa() + 10.0;
189  obs.SetPressure_hPa(pres);
190  test_double(obs.Pressure_hPa(), pres, __func__, __LINE__);
191 
192  // Test Relative humidity
193  double relHumidity = obs.RelativeHumidity();
194  if (relHumidity == 0.25) {
195  relHumidity = 0.5;
196  } else {
197  relHumidity = 0.25;
198  }
199  obs.SetRelativeHumidity(relHumidity);
200  test_double(obs.RelativeHumidity(), relHumidity, __func__, __LINE__);
201 
202  // Test wavelength
203  double wavelength = obs.Wavelength_um() + 10.0;
204  obs.SetWavelength_um(wavelength);
205  test_double(obs.Wavelength_um(), wavelength, __func__, __LINE__);
206 
207  // Make sure the print statement actually does something
208  test(obs.print().size() > 0, __func__, __LINE__);
209 
210  // Reset the tolerance
211  SetDblTol(old_tol);
212  return pass();
213 }

221 bool test_CEObserver::test_get_obsCoords(void)
222 {
223  CEBody body("North Pole", 0.0, 90.0, CECoordinateType::ICRS, CEAngleType::DEGREES);
224  CEDate date(CppEphem::julian_date_J2000(), CEDateType::JD);
225  CECoordinates obs_coords = base_obs_.ObservedPosition(body, date);
226 
227  // Note that these test coordinates were derived using Astropy
228  // to ensure the values returned are correct
229  CECoordinates test_coords(359.9960211433963, 90.00137453000666,
230  CECoordinateType::OBSERVED, CEAngleType::DEGREES);
231 
232  // Make sure observed coordinates are near where we expect them to be
233  double angsep = CECoordinates::AngularSeparation(obs_coords, test_coords,
234  CEAngleType::RADIANS);
235  // Make sure the accuracy is within 0.5 arcsecond
236  if (!test_lessthan(angsep, 0.5*DAS2R, __func__, __LINE__)) {
237  std::printf("date: %f\n", date.Gregorian());
238  std::printf("dut1,x,y: %f sec, %e rad, %e rad\n", date.dut1(), date.xpolar(), date.ypolar());
239  std::printf("Obs %s", obs_coords.print().c_str());
240  std::printf("Test %s", test_coords.print().c_str());
241  std::printf("Angsep: %e deg\n", angsep * DR2D);
242  std::printf("%s", base_obs_.print().c_str());
243  }
244 
245  return pass();
246 }
