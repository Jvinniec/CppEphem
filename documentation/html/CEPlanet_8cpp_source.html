341 void CEPlanet::UpdateCoordinates(double new_jd)
342 {
343  // If no date was supplied, or if the date hasnt changed, do nothing
344  if ((new_jd < -1.0e29) || (new_jd == cached_jd_)) return ;
345 
346  /* Date has changed, so we need to recompute the coordinates of this object */
347 
348  // Compute the number of centuries since J2000 epoch
349  double T = (new_jd-CppEphem::julian_date_J2000()) / 36525.6898326 ;
350 
351  // Compute the values
352  double a = ComputeElement(semi_major_axis_au_, semi_major_axis_au_per_cent_, T) ;
353  double e = ComputeElement(eccentricity_, eccentricity_per_cent_, T) ;
354  double I = ComputeElement(inclination_deg_, inclination_deg_per_cent_, T) ;
355  double L = ComputeElement(mean_longitude_deg_, mean_longitude_deg_per_cent_, T) ;
356  double w = ComputeElement(perihelion_lon_deg_, perihelion_long_deg_per_cent_, T) ;
357  double O = ComputeElement(ascending_node_lon_deg_, ascending_node_lon_deg_per_cent_, T) ;
358 
359  // Compute the argument of perihelion (W) and mean anomoly (M)
360  double W = w - O ;
361  double M = MeanAnomaly(L, w, T) ;
362 
363  // Iterate on the eccentric anomoly
364  double E = M + e*DR2D * std::sin(M*DD2R) ;
365  double del_E(0.0) ;
366  E = EccentricAnomoly(M, e, E, del_E) ;
367 
368  // Compute the planets heliocentric coordinates in it's orbital plane
369  double x_prime = a * (std::cos(E*DD2R) - e) ;
370  double y_prime = a * std::sqrt(1.0 - (e*e)) * std::sin(E*DD2R) ;
371 // double z_prime = 0.0 ; // by definition
372 
373  // Transform the coordinates into the coordinates of the ecliptic
374  double x_ecl = x_prime * (std::cos(W*DD2R)*std::cos(O*DD2R) - std::sin(W*DD2R)*std::sin(O*DD2R)*std::cos(I*DD2R)) +
375  y_prime * (-std::sin(W*DD2R)*std::cos(O*DD2R) - std::cos(W*DD2R)*std::sin(O*DD2R)*std::cos(I*DD2R)) ;
376  double y_ecl = x_prime * (std::cos(W*DD2R)*std::sin(O*DD2R) + std::sin(W*DD2R)*std::cos(O*DD2R)*std::cos(I*DD2R)) +
377  y_prime * (-std::sin(W*DD2R)*std::sin(O*DD2R) + std::cos(W*DD2R)*std::cos(O*DD2R)*std::cos(I*DD2R)) ;
378  double z_ecl = x_prime * (std::sin(W*DD2R)*std::sin(I*DD2R)) + y_prime * (std::cos(W*DD2R)*std::sin(I*DD2R)) ;
379 
380  // Convert from the ecliptic coordinates to ICRS
381  double obl = DD2R * 23.43928 ;
382  x_icrs_ = x_ecl ;
383  y_icrs_ = y_ecl * std::cos(obl) - z_ecl * std::sin(obl) ;
384  z_icrs_ = y_ecl * std::sin(obl) + z_ecl * std::cos(obl) ;
385 
386  double x_eq(x_icrs_), y_eq(y_icrs_), z_eq(z_icrs_) ;
387  // If there is a reference object, then compute more accurate RA,DEC
388  if (reference_ != nullptr) {
389  reference_->UpdateCoordinates(new_jd) ;
390  x_eq = x_icrs_ - reference_->GetXICRS() ;
391  y_eq = y_icrs_ - reference_->GetYICRS() ;
392  z_eq = z_icrs_ - reference_->GetZICRS() ;
393  }
394 
395  // Now compute the actual coordiantes in ICRS
396  xcoord_ = std::atan2(y_eq, x_eq) ;
397  ycoord_ = M_PI_2 - std::acos(z_eq / std::sqrt(x_eq*x_eq + y_eq*y_eq + z_eq*z_eq)) ;
398 
399  // Make sure the x-coordinate is in the appropriate range
400  while (xcoord_ > M_PI*2.0) xcoord_ -= M_PI*2.0 ;
401  while (xcoord_ < 0.0) xcoord_ += M_PI*2.0 ;
402 
403  // Now that the coordinates are updated, reset the time
404  cached_jd_ = new_jd ;
405 }

415 double CEPlanet::EccentricAnomoly(double& M,
416  double& e,
417  double& En,
418  double& del_E)
419 {
420  // Note that because this method is called recursively, no
421  // variables should be created inside of this method
422 
423  // compute del_E
424  del_E = (M - (En - e*DR2D*std::sin(En*DD2R))) / (1.0-e*std::cos(En*DD2R)) ;
425 
426  // If the change is less than the tolerance, we're all good!
427  if ( std::fabs(del_E) < E_tol) {
428  return En ;
429  }
430  // If not, then do another iteration
431  else {
432  En += del_E ;
433  return EccentricAnomoly(M, e, En, del_E) ;
434  }
