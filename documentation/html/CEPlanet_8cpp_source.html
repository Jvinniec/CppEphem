489 void CEPlanet::UpdateCoordinates(double new_jd)
490 {
491  // If no date was supplied, or if the date hasnt changed, do nothing
492  if ((new_jd < -1.0e29) || (new_jd == cached_jd_)) {
493  return ;
494  } else if (algorithm_type_ == CEPlanetAlgo::JPL) {
495  Update_JPL(new_jd) ;
496  } else if (algorithm_type_ == CEPlanetAlgo::SOFA) {
497  Update_SOFA(new_jd) ;
498  }
499 
500  double x_eq(x_icrs_), y_eq(y_icrs_), z_eq(z_icrs_) ;
501  // If there is a reference object, then compute more accurate RA,DEC
502  if (reference_ != nullptr) {
503  reference_->UpdateCoordinates(new_jd) ;
504  x_eq = x_icrs_ - reference_->GetXICRS() ;
505  y_eq = y_icrs_ - reference_->GetYICRS() ;
506  z_eq = z_icrs_ - reference_->GetZICRS() ;
507  }
508 
509  // Now compute the actual coordiantes in ICRS
510  xcoord_ = std::atan2(y_eq, x_eq) ;
511  ycoord_ = M_PI_2 - std::acos(z_eq / std::sqrt(x_eq*x_eq + y_eq*y_eq + z_eq*z_eq)) ;
512 
513  // Make sure the x-coordinate is in the appropriate range
514  while (xcoord_ > M_PI*2.0) xcoord_ -= M_PI*2.0 ;
515  while (xcoord_ < 0.0) xcoord_ += M_PI*2.0 ;
516 
517  // Now that the coordinates are updated, reset the time
518  cached_jd_ = new_jd ;
519 }

561 void CEPlanet::Update_JPL(double new_jd)
562 {
563  /* Date has changed, so we need to recompute the coordinates of this object */
564 
565  // Compute the number of centuries since J2000 epoch
566  double T = (new_jd-CppEphem::julian_date_J2000()) / 36525.6898326 ;
567 
568  // Compute the values
569  double a = ComputeElement(semi_major_axis_au_, semi_major_axis_au_per_cent_, T) ;
570  double e = ComputeElement(eccentricity_, eccentricity_per_cent_, T) ;
571  double I = ComputeElement(inclination_deg_, inclination_deg_per_cent_, T) ;
572  double L = ComputeElement(mean_longitude_deg_, mean_longitude_deg_per_cent_, T) ;
573  double w = ComputeElement(perihelion_lon_deg_, perihelion_long_deg_per_cent_, T) ;
574  double O = ComputeElement(ascending_node_lon_deg_, ascending_node_lon_deg_per_cent_, T) ;
575 
576  // Compute the argument of perihelion (W) and mean anomoly (M)
577  double W = w - O ;
578  double M = MeanAnomaly(L, w, T) ;
579

582  double del_E(0.0) ;
583  E = EccentricAnomoly(M, e, E, del_E) ;

586  double x_prime = a * (std::cos(E*DD2R) - e) ;
587  double y_prime = a * std::sqrt(1.0 - (e*e)) * std::sin(E*DD2R) ;

592  y_prime * (-std::sin(W*DD2R)*std::cos(O*DD2R) - std::cos(W*DD2R)*std::sin(O*DD2R)*std::cos(I*DD2R)) ;

633 double CEPlanet::MeanAnomaly(double mean_longitude_deg,
634  double perihelion_long_deg,
635  double T)
636 {
637  double M = mean_longitude_deg - perihelion_long_deg + (b_*T*T)
638  + c_*std::cos(f_*T*DD2R) + s_*std::sin(f_*T*DD2R) ;
639  // Scale M to be in the range (-180,180)
640  while (M>180.0) M-=360.0 ;
641  while (M<-180.0) M+=360.0 ;
642 
643  return M ;

652 double CEPlanet::EccentricAnomoly(double& M,
653  double& e,
654  double& En,
655  double& del_E)
656 {
657  // Note that because this method is called recursively, no
658  // variables should be created inside of this method
659 
660  // compute del_E
661  del_E = (M - (En - e*DR2D*std::sin(En*DD2R))) / (1.0-e*std::cos(En*DD2R)) ;
662 
663  // If the change is less than the tolerance, we're all good!
664  if ( std::fabs(del_E) < E_tol) {
665  return En ;
666  }
667  // If not, then do another iteration
668  else {
669  En += del_E ;
670  return EccentricAnomoly(M, e, En, del_E) ;
671  }
