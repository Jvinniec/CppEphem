588 /**********************************************************************/
594 double CEPlanet::EarthDist_AU(const CEDate& date)
595 {
596  // Compute the position of the earth relative to the planet to account
597  // for the time delay of the planet
598  CEPlanet earth = CEPlanet::Earth();
599  earth.SetAlgorithm(Algorithm());
600  earth.UpdatePosition(date.JD());
601 
602  std::vector<double> pos = pos_icrs_;
603  double x = pos[0] - earth.GetXICRS();
604  double y = pos[1] - earth.GetYICRS();
605  double z = pos[2] - earth.GetZICRS();
606 
607  // Compute the light travel time from the planet to Earth in days
608  return std::sqrt(x*x + y*y + z*z);
609 }

619 std::vector<double> CEPlanet::PositionICRS_Obs(const CEDate& date) const
620 {
621  // Make a copy of the planet so we dont need to reset values
622  CEPlanet past(*this);
623  past.UpdatePosition(date);
624 
625  // Compute the light travel-time delay
626  double earth_dist = past.EarthDist_AU(date);
627  double delay = earth_dist / CppEphem::c_au_per_day();
628 
629  // Return the positions as observed
630  past.UpdatePosition(date.JD() - delay);
631  return past.PositionICRS();
632 }

635 /**********************************************************************/
642 CECoordinates CEPlanet::ObservedCoords(const CEDate& date,
643  const CEObserver& observer) const
644 {
645  // Update planet position
646  UpdatePosition(date);
647 
648  // Get the observer ICRS position on a given date and the time-delayed
649  // position of the planet on that date
650  std::vector<double> offset = observer.PositionICRS(date);
651  std::vector<double> pos_obs = PositionICRS_Obs(date);
652 
653  // Compute the difference in coordinates
654  for (int i=0; i<3; i++)
655  offset[i] = pos_obs[i] - offset[i];
656 
657  // Convert the offset into RA,Dec
658  double ra(0.0);
659  double dec(0.0);
660  iauC2s(&offset[0], &ra, &dec);
661 
662  // Note that the coordinate system depends on the planet
663  CECoordinateType coordsys = CECoordinateType::ICRS;
664  if (sofa_planet_id_ == 3.5) {
665  coordsys = CECoordinateType::CIRS;
666  }
667  CECoordinates coord(ra, dec, coordsys, CEAngleType::RADIANS);
668 
669  return coord.ConvertTo(CECoordinateType::OBSERVED, observer, date);
670 }

673 /**********************************************************************/
680 void CEPlanet::UpdateCoordinates(double new_jd) const
681 {
682  // If no date was supplied, or if the date hasnt changed, do nothing
683  if ((new_jd < -1.0e29) || (new_jd == cached_jd_)) return;
684 
685  // Update the positions given the current julian date
686  UpdatePosition(new_jd);
687 
688  CEDate date(new_jd, CEDateType::JD);
689  std::vector<double> pos_delayed = PositionICRS_Obs(date);
690 
691  // Now compute the actual sky coordinates in ICRS
692  iauC2s(&pos_delayed[0], &xcoord_, &ycoord_);
693 
694  // Make sure the x-coordinate is in the appropriate range
695  xcoord_ = iauAnp(xcoord_);
696 
697  // Now that the coordinates are updated, reset the time
698  cached_jd_ = new_jd ;
699 }

718 /**********************************************************************/
726 void CEPlanet::Update_SOFA(double new_jd) const
727 {
728  // Compute the TDB based on supplied jd
729  double tdb1(0.0);
730  double tdb2(0.0);
731  double mjd = new_jd - CEDate::GetMJD2JDFactor();
732  CEDate::UTC2TDB(mjd, &tdb1, &tdb2);
733 
734  // Compute the Earth relative corrections
735  double pvb[2][3];
736  double pvhe[2][3];
737  int err = iauEpv00(tdb1, tdb2, pvhe, pvb);
738 
739  // If this isn't the earth, compute the planet's heliocentric position
740  if (sofa_planet_id_ != 3.5) {
741  double pvh[2][3];
742  err = iauPlan94(tdb1, tdb2, sofa_planet_id_, pvh) ;
743 
744  // Compute the corrected factors for heliocentric -> barycentric as
745  // bary_planet = helio_planet + (bary_earth - helio_earth)
746  for (int i=0; i<2; i++) {
747  for (int j=0; j<3; j++) {
748  pvb[i][j] = pvh[i][j] + (pvb[i][j] - pvhe[i][j]);
749  }
750  }
751  }
752 
753  // If the returned error code is zero, compute the values
754  if (err == 0) {
755  // position
756  pos_icrs_ = std::vector<double>(pvb[0], pvb[0]+3);
757 
758  // velocity
759  vel_icrs_ = std::vector<double>(pvb[1], pvb[1]+3);
760  } else {
761  std::cerr << "[ERROR] failed to compute planet positions for " << Name() << ". ErrCode=" << err << "\n" ;
762  }
763 }

765 /**********************************************************************/
771 void CEPlanet::Update_JPL(double new_jd) const
772 {
773  /* Date has changed, so we need to recompute the coordinates of this object */
774 
775  // Compute the number of centuries since J2000 epoch
776  double mjd = new_jd - CEDate::GetMJD2JDFactor();
777  double tdb1;
778  double tdb2;
779  CEDate::UTC2TDB(mjd, &tdb1, &tdb2);
780  double T = (tdb1+tdb2 - CppEphem::julian_date_J2000()) / DJC;
781 
782  // Compute the values
783  double a = ComputeElement(semi_major_axis_au_, semi_major_axis_au_per_cent_, T) ;
784  double e = ComputeElement(eccentricity_, eccentricity_per_cent_, T) ;
785  double I = ComputeElement(inclination_deg_, inclination_deg_per_cent_, T) ;
786  double L = ComputeElement(mean_longitude_deg_, mean_longitude_deg_per_cent_, T) ;
787  double w = ComputeElement(perihelion_lon_deg_, perihelion_long_deg_per_cent_, T) ;
788  double O = ComputeElement(ascending_node_lon_deg_, ascending_node_lon_deg_per_cent_, T) ;
789 
790  // Compute the argument of perihelion (W) and mean anomoly (M)
791  double W = w - O ;
792  double M = MeanAnomaly(L, w, T) ;
793 
794  // Iterate on the eccentric anomoly
795  double E = M + e*DR2D * std::sin(M*DD2R) ;
796  double del_E(0.0) ;
797  E = EccentricAnomoly(M, e, E, del_E) ;
798 
799  // Compute the planets heliocentric coordinates in it's orbital plane
800  double x_prime = a * (std::cos(E*DD2R) - e) ;
801  double y_prime = a * std::sqrt(1.0 - (e*e)) * std::sin(E*DD2R) ;
802 // double z_prime = 0.0 ; // by definition
803 
804  // Transform the coordinates into the coordinates of the ecliptic
805  double x_ecl = x_prime * (std::cos(W*DD2R)*std::cos(O*DD2R) - std::sin(W*DD2R)*std::sin(O*DD2R)*std::cos(I*DD2R)) +
806  y_prime * (-std::sin(W*DD2R)*std::cos(O*DD2R) - std::cos(W*DD2R)*std::sin(O*DD2R)*std::cos(I*DD2R)) ;
807  double y_ecl = x_prime * (std::cos(W*DD2R)*std::sin(O*DD2R) + std::sin(W*DD2R)*std::cos(O*DD2R)*std::cos(I*DD2R)) +
808  y_prime * (-std::sin(W*DD2R)*std::sin(O*DD2R) + std::cos(W*DD2R)*std::cos(O*DD2R)*std::cos(I*DD2R)) ;
809  double z_ecl = x_prime * (std::sin(W*DD2R)*std::sin(I*DD2R)) + y_prime * (std::cos(W*DD2R)*std::sin(I*DD2R)) ;
810 
811  // Convert from the ecliptic coordinates to ICRS
812  double obl = DD2R * 23.43928 ;
813  pos_icrs_[0] = x_ecl ;
814  pos_icrs_[1] = y_ecl * std::cos(obl) - z_ecl * std::sin(obl) ;
815  pos_icrs_[2] = y_ecl * std::sin(obl) + z_ecl * std::cos(obl) ;
816 }

819 /**********************************************************************/
827 double CEPlanet::MeanAnomaly(double mean_longitude_deg,
828  double perihelion_long_deg,
829  double T) const
830 {
831  double M = mean_longitude_deg - perihelion_long_deg + (b_*T*T)
832  + c_*std::cos(f_*T*DD2R) + s_*std::sin(f_*T*DD2R) ;
833  // Scale M to be in the range (-180,180)
834  while (M>180.0) M-=360.0 ;
835  while (M<-180.0) M+=360.0 ;
836 
837  return M ;
838 }

840 /**********************************************************************/
848 double CEPlanet::EccentricAnomoly(double& M,
849  double& e,
850  double& En,
851  double& del_E) const
852 {
853  // Note that because this method is called recursively, no
854  // variables should be created inside of this method
855 
856  // compute del_E
857  del_E = (M - (En - e*DR2D*std::sin(En*DD2R))) / (1.0-e*std::cos(En*DD2R)) ;
858 
859  // If the change is less than the tolerance, we're all good!
860  if ( std::fabs(del_E) < E_tol) {
861  return En ;
862  }
863  // If not, then do another iteration
864  else {
865  En += del_E ;
866  return EccentricAnomoly(M, e, En, del_E) ;
867  }
868 }
