99 std::map<std::string, double> parseoptions(int argc, char** argv, const struct option* longopts)
100 {
101  // Setup the default parameters
102  std::map<std::string, double> options = defaultoptions() ;
103 
104  int c;
105 
106  while (1)
107  {
108  /* getopt_long stores the option index here. */
109  int option_index = 0;
110 
111  c = getopt_long (argc, argv, "hX:Y:e:L:B:j:p:t:r:w:d:x:y:",
112  longopts, &option_index);
113 
114  /* Detect the end of the options. */
115  if (c == -1) break;
116 
117  switch (c)
118  {
119  case 0:
120  /* If this option set a flag, do nothing else now. */
121  if (longopts[option_index].flag != 0)
122  break;
123 
124  // Set the option
125  options[longopts[option_index].name] = std::stod(optarg) ;
126  break;
127 
128  case 'h':
129  // Print the help message and quit
130  Print_Help() ;
131  return std::map<std::string, double>() ;
132  case 'X': // Assign observer's Longitude
133  options["longitude"] = std::stod(optarg) ;
134  break;
135  case 'Y': // Assign observer's Latitude
136  options["latitude"] = std::stod(optarg) ;
137  break;
138  case 'e': // Assign observer's elevation
139  options["elevation"] = std::stod(optarg) ;
140  break;
141  case 'L': // Assign galactic longitude coordinate
142  options["glon"] = std::stod(optarg) ;
143  break;
144  case 'B': // Assign galactic latitude coordinate
145  options["glat"] = std::stod(optarg) ;
146  break;
147  case 'j': // Assign julian date
148  options["juliandate"] = std::stod(optarg) ;
149  break;
150  case 'p':
151  options["pressure"] = std::stod(optarg) ;
152  break;
153  case 't':
154  options["temperature"] = std::stod(optarg) ;
155  break;
156  case 'r':
157  options["humidity"] = std::stod(optarg) ;
158  break;
159  case 'w':
160  options["wavelength"] = std::stod(optarg) ;
161  break;
162  case 'd':
163  options["dut1"] = std::stod(optarg) ;
164  break;
165  case 'x':
166  options["xpolar"] = std::stod(optarg) ;
167  break;
168  case 'y':
169  options["ypolar"] = std::stod(optarg) ;
170  break;
171  case '?':
172  /* getopt_long already printed an error message. */
