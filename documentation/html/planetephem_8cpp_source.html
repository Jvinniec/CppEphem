115 void PrintEphemeris(CEObservation& obs, double duration, double step_size)
116 {
117  // Lets get the objects in obs
118  CEDate* date = obs.Date() ;
119  CEPlanet* planet = dynamic_cast<CEPlanet*>( obs.Body() ) ;
120  CEObserver* observer = obs.Observer() ;
121  std::vector<double> localtime = CETime::TimeDbl2Vect(observer->Date()->GetTime(observer->UTCOffset())) ;
122 
123  // Print some information about the observer
124  std::vector<double> long_hms = CECoordinates::GetDMS( observer->Longitude_Deg() ) ;
125  std::vector<double> lat_hms = CECoordinates::GetDMS( observer->Latitude_Deg() ) ;
126  std::printf("\n") ;
127  std::printf("= OBSERVER ===================\n");
128  std::printf(" Longitude: %+4dd %02dm %4.1fs\n", int(long_hms[0]), int(long_hms[1]), long_hms[2]) ;
129  std::printf(" Latitude : %+2dd %02dm %4.1fs\n", int(lat_hms[0]), int(lat_hms[1]), lat_hms[2]) ;
130  std::printf(" Elevation: %f m \n", observer->Elevation_m()) ;
131  std::printf(" LocalTime: %02d:%02d:%04.1f\n\n", int(localtime[0]), int(localtime[1]), localtime[2]+localtime[3]) ;
132 
133  // Print some basic information regarding the planet itself
134  std::printf("= PLANET =====================\n") ;
135  std::printf(" Name : %s\n", planet->Name().c_str()) ;
136  std::printf(" Mass : %e kg\n", planet->Mass_kg()) ;
137  std::printf(" Radius: %f km\n", planet->Radius_m()/1000.0) ;
138  std::printf(" Albedo: %f\n\n", planet->Albedo()) ;
139 
140  std::vector<double> ra, dec ;
141 
142  // Now do the stuff we actually want
143  std::printf(" JD LOCAL RA DEC Az Alt \n") ;
144  std::printf(" =======================================================================\n") ;
145  int max_steps = int(duration/step_size) ;
146  for (int s=0; s<=max_steps; s++) {
147 
148  ra = CECoordinates::GetHMS( planet->XCoordinate_Deg() );
149  dec = CECoordinates::GetDMS( planet->YCoordinate_Deg() );
150 
151  std::printf(" %11.2f %08.1f %2.0fh %2.0fm %4.1fs %+3.0fd %2.0fm %4.1fs %8.3f %+7.3f\n",
152  double(*date), date->GetTime(observer->UTCOffset()),
153  ra[0], ra[1], ra[2],
154  dec[0], dec[1], dec[2],
155  obs.GetAzimuth_Deg(),
156  90.0-obs.GetZenith_Deg()) ;
157 
158  // Update the date
159  date->SetDate(*date + step_size/(60.0*24.0)) ;
160  }
161 
162  std::printf(" -------------------------------------------------------------------\n") ;
163 }
