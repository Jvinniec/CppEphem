41 int main(int argc, char** argv)
42 {
43  // Get the command line options
44  CEExecOptions opts = DefineOpts() ;
45  if (opts.ParseCommandLine(argc,argv)) return 0;
46 
47  // Turn on interpolation for correction terms
48  CppEphem::CorrectionsInterp(true);
49 
50  // Create the observed object
51  CEPlanet planet = GetPlanet(opts.AsInt("planet")) ;
52  if (planet.Name().empty()) return 0 ;
53 
54  // Set the algorithm
55  std::string method = opts.AsString("method");
56  std::transform(method.begin(), method.end(), method.begin(), ::tolower);
57  if (method == "sofa") {
58  planet.SetAlgorithm(CEPlanetAlgo::SOFA);
59  } else if (method == "jpl") {
60  planet.SetAlgorithm(CEPlanetAlgo::JPL);
61  } else {
62  throw CEException::invalid_value("planetephem.cpp",
63  "[ERROR] Invalid planet algorithm: " + method);
64  }
65 
66  // Create the date object
67  CEDate date(opts.AsDouble("startJD"), CEDateType::JD) ;
68 
69  // Create the observer
70  CEObserver observer(opts.AsDouble("longitude"),
71  opts.AsDouble("latitude"),
72  opts.AsDouble("elevation"),
73  CEAngleType::DEGREES) ;
74  observer.SetPressure_hPa(opts.AsDouble("pressure"));
75  observer.SetTemperature_C(opts.AsDouble("temperature"));
76  observer.SetRelativeHumidity(opts.AsDouble("humidity"));
77  observer.SetWavelength_um(opts.AsDouble("wavelength"));
78  observer.SetUTCOffset(CETime::SystemUTCOffset_hrs());
79 
80  CEObservation coords(&observer, &planet, &date) ;
81 
82  // Now list the information requested
83  PrintEphemeris(coords, opts.AsDouble("duration"), opts.AsDouble("step")) ;
84 
85  return 0;
86 }

157 void PrintEphemeris(CEObservation& obs,
158  const double& duration,
159  const double& step_size)
160 {
161  // Lets get the objects in obs
162  CEDate* date = obs.Date();
163  CEPlanet* planet = dynamic_cast<CEPlanet*>( obs.Body() );
164  CEObserver* observer = obs.Observer();
165  std::vector<double> localtime = CETime::TimeDbl2Vect(date->GetTime(observer->UTCOffset()));
166 
167  // Print some information about the observer
168  std::vector<double> long_hms = CECoordinates::GetDMS( observer->Longitude_Deg(),
169  CEAngleType::DEGREES );
170  std::vector<double> lat_hms = CECoordinates::GetDMS( observer->Latitude_Deg(),
171  CEAngleType::DEGREES );
172  std::printf("\n") ;
173  std::printf("= OBSERVER ===================\n");
174  std::printf(" Longitude: %+4dd %02dm %4.1fs\n", int(long_hms[0]), int(long_hms[1]), long_hms[2]+long_hms[3]);
175  std::printf(" Latitude : %+3dd %02dm %4.1fs\n", int(lat_hms[0]), int(lat_hms[1]), lat_hms[2]+lat_hms[3]);
176  std::printf(" Elevation: %f m \n", observer->Elevation_m());
177  std::printf(" Pressure : %f hPa\n", observer->Pressure_hPa());
178  std::printf(" Temp : %f Celsius\n", observer->Temperature_C());
179  std::printf(" Humidity : %f %%\n", 100.0*observer->RelativeHumidity());
180  std::printf(" Wavelen. : %f micrometers\n", observer->Wavelength_um());
181  std::printf(" LocalTime: %02d:%02d:%04.1f\n", int(localtime[0]), int(localtime[1]), localtime[2]+localtime[3]);
182  std::printf(" UTCOffset: %2d hrs\n\n", int(observer->UTCOffset()));
183 
184  // Print some basic information regarding the planet itself
185  std::printf("= PLANET =====================\n");
186  std::printf(" Name : %s\n", planet->Name().c_str());
187  std::printf(" Mass : %e kg\n", planet->Mass_kg());
188  std::printf(" Radius: %f km\n", planet->Radius_m()/1000.0);
189  std::printf(" Albedo: %f\n\n", planet->Albedo());
190 
191  std::vector<double> ra;
192  std::vector<double> dec;
193 
194  // Now do the stuff we actually want
195  std::printf("= NOTES ======================\n");
196  std::printf(" * RA,DEC represent apparent ICRS coordinates for the observer\n");
197  std::printf(" * Coordinates should be considered dubious at small altitudes\n");
198  std::printf("\n");
199  std::printf(" JD LOCAL RA (appar.) DEC (appar.) Az Alt \n") ;
200  std::printf(" =======================================================================\n") ;
201  int max_steps = int(duration/step_size);
202  CECoordinates obs_coords;
203  CECoordinates appar_coords;
204  for (int s=0; s<=max_steps; s++) {
205 
206  // Get the observed coordinates
207  obs_coords.SetCoordinates(obs.GetAzimuth_Rad(), obs.GetZenith_Rad(),
208  CECoordinateType::OBSERVED, CEAngleType::RADIANS);
209  // Convert to RA,DEC
210  appar_coords = obs_coords.ConvertToICRS(date->JD(),
211  observer->Longitude_Rad(),
212  observer->Latitude_Rad(),
213  observer->Elevation_m(),
214  observer->Pressure_hPa(),
215  observer->Temperature_C(),
216  observer->RelativeHumidity(),
217  date->dut1(),
218  date->xpolar(),
219  date->ypolar(),
220  observer->Wavelength_um());
221  // Update the coordiantes of the planet
222  ra = CECoordinates::GetHMS( appar_coords.XCoordinate_Rad(), CEAngleType::RADIANS);
223  dec = CECoordinates::GetDMS( appar_coords.YCoordinate_Rad(), CEAngleType::RADIANS);
224 
225  std::printf(" %11.2f %08.1f %2.0fh %2.0fm %4.1fs %+3.0fd %2.0fm %4.1fs %8.3f %+7.3f\n",
226  double(*date), date->GetTime(observer->UTCOffset()),
227  ra[0], ra[1], ra[2] + ra[3],
228  dec[0], dec[1], dec[2] + dec[3],
229  obs_coords.XCoordinate_Deg(),
230  90.0-obs_coords.YCoordinate_Deg());
231 
232  // Update the date
233  date->SetDate(date->JD() + step_size/(60.0*24.0), CEDateType::JD);
234  }
235 
236  std::printf(" -------------------------------------------------------------------\n");
237 }
