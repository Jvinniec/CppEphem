140 CEAngle CESkyCoord::AngularSeparation(const CESkyCoord& coords1,
141  const CESkyCoord& coords2)
142 {
143  // Make sure the coordinates are in the same frame
144  if (coords1.GetCoordSystem() != coords2.GetCoordSystem()) {
145  throw CEException::invalid_value("CESkyCoord::AngularSeparation(CESkyCoord&, CESkyCoord&)",
146  "Supplied coordinates are in different frames");
147  }
148 
149  // Get the appropriate Y-Coordinates
150  double y1 = coords1.YCoord().Rad();
151  double y2 = coords2.YCoord().Rad();
152  if (coords1.GetCoordSystem() == CESkyCoordType::OBSERVED) {
153  y1 = M_PI_2 - y1;
154  y2 = M_PI_2 - y2;
155  }
156 
157  // Convert the second coordinates to be the same type as the first set of coordinates
158  return AngularSeparation(coords1.XCoord(), y1,
159  coords2.XCoord(), y2);
160 }

201 void CESkyCoord::CIRS2ICRS(const CESkyCoord& in_cirs,
202  CESkyCoord* out_icrs,
203  const CEDate& date)
204 {
205  double eo; // Equation of the origins
206 
207  // Compute the TDB date
208  double tdb1(0.0);
209  double tdb2(0.0);
210  CEDate::UTC2TDB(date.MJD(), &tdb1, &tdb2);
211 
212  // Use the SOFA library to compute the new date
213  double return_ra(0.0);
214  double return_dec(0.0);
215  iauAtic13(in_cirs.XCoord(), in_cirs.YCoord(),
216  tdb1, tdb2,
217  &return_ra, &return_dec, &eo);
218 
219  // Subtract the eo from RA if J2000 coordinates are desired
220  //*return_ra -= eo;
221 
222  out_icrs->SetCoordinates(CEAngle(return_ra), CEAngle(return_dec),
223  CESkyCoordType::ICRS);
224  return;

269 
270  // Call the necessary sofa method
271  double az(0.0);
272  double zen(0.0);
273  int err_code = iauAtio13(in_cirs.XCoord(),
274  in_cirs.YCoord(),
275  CEDate::GetMJD2JDFactor(), date.MJD(),

286  &temp_hour_angle,
287  &temp_ra, &temp_dec);
288 
289  // Handle the error code
290  if (err_code == -1) {
291  throw CEException::sofa_error("CESkyCoord::CIRS2Observed",
292  "iauAtio13", -1,
293  "SOFA method was passed an unacceptable date");
294  }
295

303  CESkyCoordType::CIRS);
304  }

306  // Set the hour angle
