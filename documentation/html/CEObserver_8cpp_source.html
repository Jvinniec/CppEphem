287 void CEObserver::UpdatePosVel(const CEDate& date) const
288 {
289  // Check if the cached date has changed
290  if (date.MJD() != cache_date_) {
291 
292  // Convert UTC to UT1 and TDB
293  double ut11(0.0);
294  double ut12(0.0);
295  double tdb1(0.0);
296  double tdb2(0.0);
297  double tt1(0.0);
298  double tt2(0.0);
299  CEDate::UTC2UT1(date.MJD(), &ut11, &ut12);
300  CEDate::UTC2TDB(date.MJD(), &tdb1, &tdb2);
301  CEDate::UTC2TT(date.MJD(), &tt1, &tt2);
302 
303  // Compute the Earth rotation angle at UT1
304  double theta = iauEra00(ut11, ut12);
305  double sp = iauSp00(tt1, tt2);
306 
307  // Vectors for intermediate position/velocities
308  double pvc[2][3]; // CIRS pos,vel
309 
310  // Get the position and velocity values in CIRS
311  iauPvtob(longitude_,
312  latitude_,
313  elevation_m_,
314  date.xpolar(),
315  date.ypolar(),
316  sp,
317  theta,
318  pvc);
319 
320  // Earth centric distance/velocity in ICRS (AU and AU/day)
321  double ebpv[2][3];
322  double ehpv[2][3];
323  iauEpv00(tdb1, tdb2, ehpv, ebpv);
324 
325  // Get the CIRS and ICRS pos,vel
326  double mps_apd = CppEphem::sec_per_day() / CppEphem::m_per_au();
327  for (int i=0; i<3; i++) {
328  // CIRS
329  pos_cirs_[i] = pvc[0][i] / CppEphem::m_per_au();
330  vel_cirs_[i] = pvc[1][i] * mps_apd;
331 
332  // ICRS (Earth barycenter + CIRS offset)
333  pos_icrs_[i] = ebpv[0][i] + pos_cirs_[i];
334  vel_icrs_[i] = ebpv[1][i] + vel_cirs_[i];
335  }
336  }
337  return;
338 }
