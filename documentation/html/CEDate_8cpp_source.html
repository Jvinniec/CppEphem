49 void CEDate::SetDate(double date, CEDateType time_format)
50 {
51  // Fill the internal date storage objects based on the format of the input "date"
52  if (time_format == CEDateType::JD) {
53  // Save information based on julian date input
54  julian_date_ = date ;
55  mod_julian_date_ = JD2MJD(date) ;
56  gregorian_date_ = JD2Gregorian(date) ;
57  gregorian_date_vect_ = JD2GregorianVect(date) ;
58  } else if (time_format == CEDateType::MJD) {
59  // Save information based on modified julian date input
60  julian_date_ = MJD2JD(date) ;
61  mod_julian_date_ = date ;
62  gregorian_date_ = MJD2Gregorian(date) ;
63  gregorian_date_vect_ = MJD2GregorianVect(date) ;
64  } else if (time_format == CEDateType::GREGORIAN) {
65  // Save information based on gregorian date input
66  julian_date_ = Gregorian2JD(date) ;
67  mod_julian_date_ = Gregorian2MJD(date) ;
68  gregorian_date_ = date ;
69  gregorian_date_vect_ = Gregorian2GregorianVect(date) ;
70  }
71 }

104 // Uses the SOFA function "iauJd2cal"
105 std::vector<double> CEDate::JD2GregorianVect(double jd)
106 {
107  std::vector<double> full_gregorian_vect(4,0.0) ;
108  std::vector<int> gregorian_date(3,0.0) ;
109 
110  // Try to fill the gregorian_date vector
111  if (iauJd2cal(jd, 0, &gregorian_date[0], &gregorian_date[1],
112  &gregorian_date[2], &full_gregorian_vect[3])) {
113  std::cerr << "[WARNING] CEDate::JD2GregorianVect() :: Bad date (" << jd << ")!" << std::endl ;
114  return std::vector<double>(4,0.0) ;
115  } else {
116  // Fill the double vector from the integer values
117  full_gregorian_vect[0] = 1.0*gregorian_date[0] ;
118  full_gregorian_vect[1] = 1.0*gregorian_date[1] ;
119  full_gregorian_vect[2] = 1.0*gregorian_date[2] ;
120  }
121 
122  return full_gregorian_vect ;
123 }

164 // Calls the SOFA "iauCal2jd" function
165 double CEDate::GregorianVect2JD(std::vector<double> gregorian)
166 {
167  // The following stores the two values needed to get the julian date
168  double mjd_factor(0.0), mjd(0.0) ;
169 
170  // Now use the sofa function for converting to julian date
171  int err_code = iauCal2jd(int(gregorian[0]), int(gregorian[1]),
172  int(gregorian[2]), &mjd_factor, &mjd) ;
173  if (err_code) {
174  std::cerr << "[WARNING] CEDate::GregorianVect2JD() :: Bad ";
175  if (err_code == -1) {
176  std::cerr << "year (" << gregorian[0] << "). Make sure the year is larger than -4800." << std::endl ;
177  return 0.0 ;
178  } else if (err_code == -2) {
179  std::cerr << "month (" << gregorian[1] << "). Make sure this number is in the range 1 -> 12." << std::endl ;
180  return 0.0 ;
181  } else if (err_code == -3) {
182  std::cerr << "day (" << gregorian[2] << ")" << std::endl ;
183  // Note that in this case, a julian date was still calculated
184  // so we can go ahead and return it
185  }
186  }
187 
188  // Return the sum of the two filled date values which represents
189  // the Julian date
190  return mjd_factor + mjd + gregorian[3] ;
191 }

284 std::vector<double> CEDate::Gregorian2GregorianVect(double gregorian)
285 {
286  // Create a vector to hold the information
