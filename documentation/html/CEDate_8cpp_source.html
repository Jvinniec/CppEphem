163 std::vector<double> CEDate::JD2GregorianVect(double jd)
164 {
165  std::vector<double> full_gregorian_vect(4,0.0) ;
166  std::vector<int> gregorian_date(3,0.0) ;
167 
168  // Try to fill the gregorian_date vector
169  if (iauJd2cal(jd, 0, &gregorian_date[0], &gregorian_date[1],
170  &gregorian_date[2], &full_gregorian_vect[3])) {
171  std::cerr << "[WARNING] CEDate::JD2GregorianVect() :: Bad date (" << jd << ")!" << std::endl ;
172  return std::vector<double>(4,0.0) ;
173  } else {
174  // Fill the double vector from the integer values

246 double CEDate::GregorianVect2JD(std::vector<double> gregorian)
247 {
248  // The following stores the two values needed to get the julian date
249  double mjd_factor(0.0), mjd(0.0) ;
250 
251  // Now use the sofa function for converting to julian date
252  int err_code = iauCal2jd(int(gregorian[0]), int(gregorian[1]),
253  int(gregorian[2]), &mjd_factor, &mjd) ;
254  if (err_code) {
255  std::cerr << "[WARNING] CEDate::GregorianVect2JD() :: Bad ";
256  if (err_code == -1) {
257  std::cerr << "year (" << gregorian[0] << "). Make sure the year is larger than -4800." << std::endl ;
258  return 0.0 ;
259  } else if (err_code == -2) {
260  std::cerr << "month (" << gregorian[1] << "). Make sure this number is in the range 1 -> 12." << std::endl ;
261  return 0.0 ;
262  } else if (err_code == -3) {
263  std::cerr << "day (" << gregorian[2] << ")" << std::endl ;
264  // Note that in this case, a julian date was still calculated
265  // so we can go ahead and return it
266  }
267  }
268 
269  // Return the sum of the two filled date values which represents
270  // the Julian date
271  return mjd_factor + mjd + gregorian[3] ;
272 }

378 std::vector<double> CEDate::Gregorian2GregorianVect(double gregorian)
379 {
380  // Create a vector to hold the information
381  std::vector<double> gregorian_vect(4,0.0) ;
382 
383  // Get the day fraction
384  gregorian_vect[3] = gregorian - std::floor(gregorian) ;
385  // Get the day of the month
386  gregorian_vect[2] = int(std::floor(gregorian)) % 100 ;
387  // Get the month
388  gregorian_vect[1] = int(std::floor(gregorian - gregorian_vect[2]))/100 % 100 ;
389  // Get the year
390  gregorian_vect[0] = (std::floor(gregorian - gregorian_vect[2] - gregorian_vect[1])/10000) ;
