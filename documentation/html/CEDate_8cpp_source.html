150 std::vector<double> CEDate::JD2GregorianVect(double jd)
151 {
152  std::vector<double> full_gregorian_vect(4,0.0) ;
153  std::vector<int> gregorian_date(3,0.0) ;
154 
155  // Try to fill the gregorian_date vector
156  if (iauJd2cal(jd, 0, &gregorian_date[0], &gregorian_date[1],
157  &gregorian_date[2], &full_gregorian_vect[3])) {
158  std::cerr << "[WARNING] CEDate::JD2GregorianVect() :: Bad date (" << jd << ")!" << std::endl ;
159  return std::vector<double>(4,0.0) ;
160  } else {
161  // Fill the double vector from the integer values
162  full_gregorian_vect[0] = 1.0*gregorian_date[0] ;
163  full_gregorian_vect[1] = 1.0*gregorian_date[1] ;
164  full_gregorian_vect[2] = 1.0*gregorian_date[2] ;
165  }

218 double CEDate::GregorianVect2JD(std::vector<double> gregorian)
219 {
220  // The following stores the two values needed to get the julian date
221  double mjd_factor(0.0), mjd(0.0) ;
222 
223  // Now use the sofa function for converting to julian date
224  int err_code = iauCal2jd(int(gregorian[0]), int(gregorian[1]),
225  int(gregorian[2]), &mjd_factor, &mjd) ;
226  if (err_code) {
227  std::cerr << "[WARNING] CEDate::GregorianVect2JD() :: Bad ";
228  if (err_code == -1) {
229  std::cerr << "year (" << gregorian[0] << "). Make sure the year is larger than -4800." << std::endl ;
230  return 0.0 ;
231  } else if (err_code == -2) {
232  std::cerr << "month (" << gregorian[1] << "). Make sure this number is in the range 1 -> 12." << std::endl ;
233  return 0.0 ;
234  } else if (err_code == -3) {
235  std::cerr << "day (" << gregorian[2] << ")" << std::endl ;
236  // Note that in this case, a julian date was still calculated
237  // so we can go ahead and return it
238  }
239  }
240 
241  // Return the sum of the two filled date values which represents
242  // the Julian date
243  return mjd_factor + mjd + gregorian[3] ;
244 }

342 std::vector<double> CEDate::Gregorian2GregorianVect(double gregorian)
343 {
344  // Create a vector to hold the information
345  std::vector<double> gregorian_vect(4,0.0) ;
346 
347  // Get the day fraction
348  gregorian_vect[3] = gregorian - std::floor(gregorian) ;
349  // Get the day of the month
350  gregorian_vect[2] = int(std::floor(gregorian)) % 100 ;
351  // Get the month
352  gregorian_vect[1] = int(std::floor(gregorian - gregorian_vect[2]))/100 % 100 ;
353  // Get the year
354  gregorian_vect[0] = (std::floor(gregorian - gregorian_vect[2] - gregorian_vect[1])/10000) ;
355 
356  return gregorian_vect ;
