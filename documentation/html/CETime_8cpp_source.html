80 double CETime::CurrentUTC()
81 {
82  // Construct time_t of this moment
83  std::chrono::system_clock::time_point now(std::chrono::system_clock::now());
84  std::time_t now_t(std::chrono::system_clock::to_time_t(
85  std::chrono::system_clock::time_point(now)));
86 
87  // Get the time at midnight
88  struct tm midnight = *gmtime(&now_t) ;
89  midnight.tm_hour = 0 ;
90  midnight.tm_min = 0 ;
91  midnight.tm_sec = 0 ;
92 
93  // Midnight object
94  auto millisec_start(std::chrono::system_clock::from_time_t(mktime(&midnight)));
95 
96  double start = std::chrono::duration_cast<std::chrono::microseconds>(millisec_start.time_since_epoch()).count()/1000000.0 ;
97  double stop = std::chrono::duration_cast<std::chrono::microseconds>(now.time_since_epoch()).count()/1000000.0 ;
98 
99  // Return seconds and fractions of a second since midnight
100  return stop-start-midnight.tm_gmtoff ;
101 }

215 std::vector<double> CETime::TimeDbl2Vect(double time)
216 {
217  // Create a vector to hold the information
218  std::vector<double> time_vect(4,0.0) ;
219 
220  // Get the seconds fraction
221  time_vect[3] = time - std::floor(time) ;
222  // Get the seconds value
223  time_vect[2] = int(std::floor(time)) % 100 ;
224  // Get the minutes
225  time_vect[1] = int(std::floor(time - time_vect[2]))/100 % 100 ;
226  // Get the hours
227  time_vect[0] = (std::floor(time - time_vect[2] - time_vect[1])/10000) ;
228 
229  return time_vect ;
230 }

240 double CETime::TimeSec2Time(double seconds)
241 {
242  double fracsec = seconds - std::floor(seconds) ; // Fractions of a second
243  double sec = int(std::floor(seconds)) % 60 ; // Whole seconds
244  double min = int(std::floor(seconds-sec)/60) % 60 ; // Whole minutes
245  double hrs = int(std::floor(seconds-sec)/60)/60 ; // Whole hours
246  return (hrs*10000) + (min*100) + sec + fracsec ; // Formatted double (HHMMSS.S)
247 }
