80 double CETime::CurrentUTC()
81 {
82  auto now(std::chrono::system_clock::now());
83  auto seconds_since_epoch(std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()));
84 
85  // Construct time_t using 'seconds_since_epoch' rather than 'now' since it is
86  // implementation-defined whether the value is rounded or truncated.
87  std::time_t now_t(std::chrono::system_clock::to_time_t(
88  std::chrono::system_clock::time_point(seconds_since_epoch)));
89 
90  // Get the time at midnight
91  struct tm midnight = *gmtime(&now_t) ;
92  midnight.tm_hour = 0 ;
93  midnight.tm_min = 0 ;
94  midnight.tm_sec = 0 ;
95 
96  // Return seconds and fractions of a second since midnight
97  return difftime(now_t, mktime(&midnight)) + (now.time_since_epoch() - seconds_since_epoch).count();
98 }

212 std::vector<double> CETime::TimeDbl2Vect(double time)
213 {
214  // Create a vector to hold the information
215  std::vector<double> time_vect(4,0.0) ;
216 
217  // Get the seconds fraction
218  time_vect[3] = time - std::floor(time) ;
219  // Get the seconds value
220  time_vect[2] = int(std::floor(time)) % 100 ;
221  // Get the minutes
222  time_vect[1] = int(std::floor(time - time_vect[2]))/100 % 100 ;
223  // Get the hours
224  time_vect[0] = (std::floor(time - time_vect[2] - time_vect[1])/10000) ;
225

237 double CETime::TimeSec2Time(double seconds)
238 {
239  double fracsec = seconds - std::floor(seconds) ; // Fractions of a second
240  double sec = int(std::floor(seconds)) % 60 ; // Whole seconds
241  double min = int(std::floor(seconds-sec)/60) % 60 ; // Whole minutes
242  double hrs = int(std::floor(seconds-sec)/60)/60 ; // Whole hours
243  return (hrs*10000) + (min*100) + sec + fracsec ; // Formatted double (HHMMSS.S)
244 }
