112 bool CEObservation::UpdateCoordinates()
113 {
114  // Get the coordinate system of the body being observed
115  CECoordinateType body_coords = body_->GetCoordSystem() ;
116 
117  // Compute the observed coordinates based on the coordinates of "body_"
118  if (body_coords == CECoordinateType::CIRS) {
119  // Convert from CIRS -> Observed
120  CEBody::CIRS2Observed(body_->XCoordinate_Rad(*observer_->Date()),
121  body_->YCoordinate_Rad(*observer_->Date()),
122  &cached_azimuth_, &cached_zenith_,
123  *observer_,
124  CEAngleType::RADIANS,
125  observer_->Wavelength_um(),
126  &cached_apparentxcoord_,
127  &cached_apparentycoord_,
128  &cached_hour_angle_) ;
129  } else if (body_coords == CECoordinateType::GALACTIC) {
130  // Convert Galactic -> Observerd
131  CEBody::Galactic2Observed(body_->XCoordinate_Rad(*observer_->Date()),
132  body_->YCoordinate_Rad(*observer_->Date()),
133  &cached_azimuth_, &cached_zenith_,
134  *observer_,
135  CEAngleType::RADIANS,
136  observer_->Wavelength_um(),
137  &cached_apparentxcoord_,
138  &cached_apparentycoord_,
139  &cached_hour_angle_) ;
140  } else if (body_coords == CECoordinateType::ICRS) {
141  // Convert CIRS -> Observed
142  CEBody::ICRS2Observed(body_->XCoordinate_Rad(*observer_->Date()),
143  body_->YCoordinate_Rad(*observer_->Date()),
144  &cached_azimuth_, &cached_zenith_,
145  *observer_,
146  CEAngleType::RADIANS,
147  observer_->Wavelength_um(),
148  &cached_apparentxcoord_,
149  &cached_apparentycoord_,
150  &cached_hour_angle_) ;
151  } else {
152  return false ;
153  }
154 
155  // Update the cached_date_ object
156  cached_date_ = *date_ ;
157 
158  return true ;
159 }
